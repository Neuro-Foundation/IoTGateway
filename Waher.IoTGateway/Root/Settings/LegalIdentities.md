JoinResidence(Identity):=
(
	First:=true;
	v:=[Identity.ADDR, Identity.ADDR2, Identity.ZIP, Identity.AREA, Identity.CITY, Identity.REGION, Identity.COUNTRY];
	s:="";
	foreach s2 in v do
	(
		if !empty(s2) then
		(
			if First then
				First:=false
			else
				s+=", ";
			
			s+=MarkdownEncode(s2)
		)
	);
	s
);

JoinOther(Properties[],Attachments[]):=
(
	First:=true;
	s:="";

	(
		Name:=Property.Name;
		if (Name!="FIRST" and 
			Name!="MIDDLE" and

		(
			if First then
				First:=false

				s+=", ";

	);

		)
	);
